### 1.3 识别的性能瓶颈

#### 1.3.1 服务端瓶颈

1. **数据库写入瓶颈**
   - 现状：单条插入设备数据表
   - 问题：5000-10000设备并发写入时性能下降
   - 影响：数据延迟、丢失风险

2. **MQTT订阅管理**
   - 现状：动态订阅/取消订阅
   - 问题：设备量大时订阅管理开销大
   - 影响：连接不稳定

3. **预聚合计算**
   - 现状：定时任务聚合
   - 问题：实时性不足
   - 影响：图表数据延迟

#### 1.3.2 客户端瓶颈

1. **UI渲染卡顿**
   - 现状：设备列表全量更新
   - 问题：1000-2000设备同时更新时界面卡顿
   - 影响：用户体验差

2. **图表数据查询**
   - 现状：实时查询数据库
   - 问题：7天在线率、24小时功率等查询慢
   - 影响：图表加载慢

3. **区域筛选性能**
   - 现状：遍历所有设备过滤
   - 问题：大数据量时筛选慢
   - 影响：操作响应慢

4. **内存占用**
   - 现状：设备数据全量加载到内存
   - 问题：设备量大时内存占用高
   - 影响：系统稳定性

**3. 批量写入优化**
```python
class BatchDatabaseWriter:
    """批量数据库写入器"""
    
    def __init__(self, batch_size=100, flush_interval=1):
        self.batch_size = batch_size  # 每批写入数量
        self.flush_interval = flush_interval  # 强制刷新间隔(秒)
        self.buffer = []
        self.lock = threading.Lock()
        
    def write(self, data):
        """写入数据到缓冲区"""
        with self.lock:
            self.buffer.append(data)
            
            # 达到批量大小立即写入
            if len(self.buffer) >= self.batch_size:
                self._flush()
    
    def _flush(self):
        """批量写入数据库"""
        if not self.buffer:
            return
            
        try:
            # 使用事务批量插入
            with self.db_pool.connection() as conn:
                with conn.cursor() as cursor:
                    # 批量插入语法
                    cursor.executemany(
                        "INSERT INTO device_data (...) VALUES (%s, %s, ...)",
                        self.buffer
                    )
                    conn.commit()
                    
            # 写入成功后清空缓冲区
            self.buffer.clear()
            
        except Exception as e:
            # 写入失败，保留数据到文件
            self._backup_to_file(self.buffer)
            raise
```

**4. 写入确认机制**
```python
class WriteAckManager:
    """写入确认管理器"""
    
    def __init__(self):
        self.pending_acks = {}  # 待确认写入
        self.ack_timeout = 5  # 确认超时(秒)
        
    def submit_write(self, data_id, data):
        """提交写入任务"""
        self.pending_acks[data_id] = {
            'data': data,
            'timestamp': time.time(),
            'retry_count': 0
        }
        
    def confirm_write(self, data_id):
        """确认写入成功"""
        if data_id in self.pending_acks:
            del self.pending_acks[data_id]
            
    def check_timeouts(self):
        """检查超时并重试"""
        now = time.time()
        for data_id, info in list(self.pending_acks.items()):
            if now - info['timestamp'] > self.ack_timeout:
                if info['retry_count'] < 3:
                    # 重试写入
                    self._retry_write(data_id, info)
                else:
                    # 超过重试次数，写入文件
                    self._write_to_file(info['data'])
                    del self.pending_acks[data_id]
```

class DataIntegrityChecker:
    """数据完整性校验器"""
    
    def verify_data_integrity(self, device_id, start_time, end_time):
        """校验指定时间段的数据完整性"""
        # 1. 获取期望数据量（根据上报间隔计算）

        # 2. 获取实际存储数据量

            self._trigger_data_recovery(device_id, start_time, end_time)

**1. 多线程消息接收**
```python
class RealTimeDataProcessor:
    """实时数据处理器"""
    
    def __init__(self, num_workers=32):
        self.num_workers = num_workers
        self.device_queues = {}  # 设备队列字典
        self.worker_pool = ThreadPoolExecutor(max_workers=num_workers)
        self.message_queue = Queue(maxsize=50000)  # 消息缓冲队列
        
    def start(self):
        """启动处理流水线"""
        # 启动消息分发线程
        for i in range(4):  # 4个分发线程
            threading.Thread(
                target=self._message_dispatcher,
                name=f"Dispatcher-{i}",
                daemon=True
            ).start()
            
        # 启动工作线程
        for i in range(self.num_workers):
            self.worker_pool.submit(self._worker_loop, i)
    
    def _message_dispatcher(self):
        """消息分发器 - 将消息路由到对应设备队列"""
        while self.running:
            try:
                topic, data = self.message_queue.get(timeout=1)
                device_id = topic.split("/")[1]
                
                # 获取或创建设备队列
                if device_id not in self.device_queues:
                    self.device_queues[device_id] = Queue(maxsize=1000)
                
                # 非阻塞放入队列（满时丢弃旧数据）
                queue = self.device_queues[device_id]
                try:
                    queue.put_nowait((topic, data))
                except Queue.Full:
                    # 丢弃最旧数据，保留最新
                    try:
                        queue.get_nowait()
                        queue.put_nowait((topic, data))
                    except:
                        pass
                        
            except Queue.Empty:
                continue
```

**3. WebSocket实时推送**
```python
class RealTimePushServer:
    """实时推送服务器 - 替代客户端轮询"""
    
    def __init__(self):
        self.clients = {}  # 连接的客户端
        self.device_subscribers = defaultdict(set)  # 设备订阅者
        
    async def handle_client(self, websocket, path):
        """处理客户端连接"""
        client_id = str(uuid.uuid4())
        self.clients[client_id] = {
            'websocket': websocket,
            'subscribed_devices': set()
        }
        
        try:
            async for message in websocket:
                data = json.loads(message)
                
                if data['type'] == 'subscribe':
                    # 订阅设备更新
                    device_ids = data['devices']
                    for dev_id in device_ids:
                        self.device_subscribers[dev_id].add(client_id)
                        self.clients[client_id]['subscribed_devices'].add(dev_id)
                        
                elif data['type'] == 'unsubscribe':
                    # 取消订阅
                    device_ids = data['devices']
                    for dev_id in device_ids:
                        self.device_subscribers[dev_id].discard(client_id)
                        self.clients[client_id]['subscribed_devices'].discard(dev_id)
                        
        except websockets.exceptions.ConnectionClosed:
            # 清理客户端连接
            self._cleanup_client(client_id)
    
    async def push_device_update(self, device_id, data):
        """推送设备更新到订阅的客户端"""
        subscribers = self.device_subscribers.get(device_id, set())
        
        message = json.dumps({
            'type': 'device_update',
            'device_id': device_id,
            'data': data,
            'timestamp': time.time()
        })
        
        # 并发推送到所有订阅者
        tasks = []
        for client_id in subscribers:
            client = self.clients.get(client_id)
            if client:
                tasks.append(
                    self._send_safe(client['websocket'], message)
                )
        
        if tasks:
            await asyncio.gather(*tasks, return_exceptions=True)
```

    def _fetch_online_rate_from_db(self, area_path, time_range):

            # 实时在线率 - 从内存缓存计算
            return self._calculate_realtime_online_rate(area_path)
        elif time_range == '7days':
            # 7天在线率 - 从预聚合表查询
            return self._query_daily_online_summary(area_path, days=7)
        
    def _calculate_realtime_online_rate(self, area_path):
        """计算实时在线率"""
        # 获取区域内的所有设备
        devices = self._get_devices_in_area(area_path)
        
        # 统计在线设备数
        online_count = 0
        for device in devices:
            if self._is_device_online(device['id']):
                online_count += 1
        
        total = len(devices)
        return {
            'online': online_count,
            'offline': total - online_count,
            'total': total,
            'rate': online_count / total if total > 0 else 0
        }

class SmartRefreshManager(QObject):
    """智能刷新管理器 - 避免频繁刷新"""
    
    refresh_needed = pyqtSignal(str)  # 需要刷新的区域
    
    def __init__(self):
        super().__init__()
        self.timers = {}
        self.pending_updates = set()
        self.lock = threading.Lock()
        
    def request_refresh(self, area, priority='normal'):
        """请求刷新"""
        with self.lock:
            self.pending_updates.add(area)
        
        # 根据优先级设置刷新延迟
        delay = 100 if priority == 'high' else 500 if priority == 'normal' else 1000
        
        # 取消之前的定时器
        if area in self.timers:
            self.timers[area].stop()
        
        # 创建新的定时器
        timer = QTimer(self)
        timer.setSingleShot(True)
        timer.timeout.connect(lambda: self._do_refresh(area))
        timer.start(delay)
        
        self.timers[area] = timer
    
    def _do_refresh(self, area):
        """执行刷新"""
        with self.lock:
            if area in self.pending_updates:
                self.pending_updates.remove(area)
        
        self.refresh_needed.emit(area)

class ConfigMonitor:
    """配置监控器 - 监控配置变化"""
    
    def __init__(self):
        self.callbacks = {}
        self.last_values = {}
    
    def watch(self, section: str, key: str, callback):
        """监听配置变化"""
        config_key = f"{section}.{key}"
        if config_key not in self.callbacks:
            self.callbacks[config_key] = []
        self.callbacks[config_key].append(callback)
        
        # 记录初始值
        config = getattr(config_manager, section)
        self.last_values[config_key] = getattr(config, key)
    
    def check_changes(self):
        """检查配置变化"""
        for config_key, callbacks in self.callbacks.items():
            section, key = config_key.split('.')
            config = getattr(config_manager, section)
            current_value = getattr(config, key)
            
            if current_value != self.last_values[config_key]:
                # 配置变化，触发回调
                for callback in callbacks:
                    try:
                        callback(current_value, self.last_values[config_key])
                    except Exception as e:
                        _log.error(f"配置变化回调失败: {e}")
                
                self.last_values[config_key] = current_value

class ConfigHotReloader:
    """配置热重载器"""
    
    def __init__(self, config_file='config.yaml'):
        self.config_file = config_file
        self.last_mtime = 0
        self.reload_timer = QTimer()
        self.reload_timer.timeout.connect(self._check_and_reload)
        self.reload_timer.start(5000)  # 每5秒检查一次
        
    def _check_and_reload(self):
        """检查并重新加载配置"""
        try:
            current_mtime = os.path.getmtime(self.config_file)
            
            if current_mtime > self.last_mtime:
                _log.info(f"配置文件已修改，重新加载: {self.config_file}")
                config_manager._load_from_file(self.config_file)
                self.last_mtime = current_mtime
                
                # 触发配置更新事件
                self._on_config_reloaded()
        
        except Exception as e:
            _log.error(f"检查配置文件失败: {e}")
    
    def _on_config_reloaded(self):
        """配置重新加载后的处理"""
        # 验证配置
        errors = ConfigValidator.validate_server_config(config_manager.server)
        if errors:
            _log.error(f"配置验证失败: {errors}")
            return
        
        # 应用新配置
        _log.info("配置已更新并生效")
